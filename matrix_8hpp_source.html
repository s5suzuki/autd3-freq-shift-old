133  static Matrix4X4 Translation(const Vector3<T>& v, const Quaternion<T>& q) {
134  Matrix4X4 transform_matrix;
135  transform_matrix(0, 0) = 1.f - 2.f * q.y() * q.y() - 2.f * q.z() * q.z();
136  transform_matrix(0, 1) = 2 * q.x() * q.y() + 2 * q.y() * q.z();
137  transform_matrix(0, 2) = 2 * q.x() * q.z() - 2 * q.w() * q.y();
138  transform_matrix(0, 3) = v.x();
139  transform_matrix(1, 0) = 2 * q.x() * q.y() - 2 * q.y() * q.z();
140  transform_matrix(1, 1) = 1 - 2 * q.x() * q.x() - 2 * q.z() * q.z();
141  transform_matrix(1, 2) = 2 * q.y() * q.z() + 2 * q.w() * q.x();
142  transform_matrix(1, 3) = v.y();
143  transform_matrix(2, 0) = 2 * q.x() * q.z() + 2 * q.w() * q.y();

145  transform_matrix(2, 2) = 1 - 2 * q.x() * q.x() - 2 * q.y() * q.y();
146  transform_matrix(2, 3) = v.z();
