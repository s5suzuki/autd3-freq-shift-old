190  Result<bool, std::string> WaitMsgProcessed(const uint8_t msg_id, const size_t max_trial) {
191  if (this->_link == nullptr || !this->_link->is_open()) return Ok(false);
192 
193  const auto num_dev = this->_devices.size();
194  const auto buffer_len = num_dev * EC_INPUT_FRAME_SIZE;
195 
196  reserve_rx();
197  for (size_t i = 0; i < max_trial; i++) {
198  auto res = this->ReadData(buffer_len);
199  if (res.is_err()) return res;
200 
201  size_t processed = 0;
202  for (size_t dev = 0; dev < num_dev; dev++) {
203  const uint8_t proc_id = _rx_data[dev * 2 + 1];
204  if (proc_id == msg_id) processed++;
205  }
206 
207  if (processed == num_dev) return Ok(true);
208 
209  auto wait = static_cast<size_t>(std::ceil(static_cast<double>(EC_TRAFFIC_DELAY) * 1000 / EC_DEVICE_PER_FRAME * static_cast<double>(num_dev)));
210  std::this_thread::sleep_for(std::chrono::milliseconds(wait));
211  }
212 
213  return Ok(false);
214  }

258  static uint8_t get_id() {
259  static std::atomic<uint8_t> id{0};

261  uint8_t expected = 0xFF;
