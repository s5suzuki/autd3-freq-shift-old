142  bool wait_msg_processed(const uint8_t msg_id, const size_t max_trial) const {
143  if (!this->_check_ack) return true;
144  const auto num_devices = this->_geometry->num_devices();
145  const auto buffer_len = num_devices * core::EC_INPUT_FRAME_SIZE;
146  for (size_t i = 0; i < max_trial; i++) {
147  this->_link->read(&_rx_buf[0], buffer_len);
148  if (core::Logic::is_msg_processed(num_devices, msg_id, &_rx_buf[0])) return true;
149 
150  auto wait = static_cast<size_t>(std::ceil(core::EC_TRAFFIC_DELAY * 1000.0 / core::EC_DEVICE_PER_FRAME * static_cast<double>(num_devices)));
151  std::this_thread::sleep_for(std::chrono::milliseconds(wait));
152  }
153 
154  return false;
